let decompte,
    minutes = parseInt(document.getElementById("minutes")),
    secondes = parseInt(document.getElementById("secondes")),
    milli_secondes = parseInt(document.getElementById("milli-secondes"));

go.addEventListener('click', function compteur(element) {
    console.log(document.getElementById('minutes').value);
    console.log(document.getElementById('secondes').value);
    //return false;
    decompte = setInterval(function () {
        let minute_text;
        if (document.getElementById('secondes').value == 0) {
            if (document.getElementById('minutes').value == 0) {
                document.getElementById('compteur').innerHTML = "Compte à rebours terminé";
                clearInterval(decompte);
                return;
            } else {
                document.getElementById('minutes').value -= 1;
                document.getElementById('secondes').value = 60;
            }
        }
        if (document.getElementById('minutes').value > 0) {
            minute_text = document.getElementById('minutes').value;
            minute_text -= 1
        } else {
            minute_text = '00';
        }
        let second_text = document.getElementById('secondes').value;
        document.getElementById('minutes').innerHTML = minute_text;
        document.getElementById('secondes').innerHTML = second_text;
    }, 1000);
});
